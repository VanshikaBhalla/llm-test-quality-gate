 Here is the JUnit 5 test class for the `BankAccount` class:

```java
package com.example;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertThrows;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

class BankAccountTest {

    private BankAccount account1;
    private BankAccount account2;

    @BeforeEach
    void setup() {
        account1 = new BankAccount("123456789", 100);
        account2 = new BankAccount("987654321", 50);
    }

    @Test
    void testGetAccountNumber() {
        assertEquals("123456789", account1.getAccountNumber());
        assertEquals("987654321", account2.getAccountNumber());
    }

    @Test
    void testGetBalance() {
        assertEquals(100, account1.getBalance(), 0.01);
        assertEquals(50, account2.getBalance(), 0.01);
    }

    @Test
    void testDeposit() {
        account1.deposit(50);
        assertEquals(150, account1.getBalance(), 0.01);
    }

    @Test
    void testWithdraw() {
        account1.withdraw(20);
        assertEquals(80, account1.getBalance(), 0.01);
    }

    @Test
    void testWithdraw_InsufficientFunds() {
        assertThrows(IllegalArgumentException.class, () -> account1.withdraw(200));
    }

    @Test
    void testTransferTo_InsufficientFunds() {
        assertThrows(IllegalArgumentException.class, () -> account1.transferTo(account2, 150));
    }

    @Test
    void testTransferTo() {
        account1.transferTo(account2, 50);
        assertEquals(50, account1.getBalance(), 0.01);
        assertEquals(100, account2.getBalance(), 0.01);
    }

}
```

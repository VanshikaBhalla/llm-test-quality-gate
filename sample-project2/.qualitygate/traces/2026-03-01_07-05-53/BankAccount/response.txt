 Here is the JUnit 5 test class for the BankAccount class:

```java
package com.example;

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.*;

class BankAccountTest {

    private BankAccount account1;
    private BankAccount account2;

    @BeforeEach
    void setUp() {
        account1 = new BankAccount("123456789", 1000);
        account2 = new BankAccount("987654321", 500);
    }

    @Test
    void testGetAccountNumber() {
        assertEquals("123456789", account1.getAccountNumber());
        assertEquals("987654321", account2.getAccountNumber());
    }

    @Test
    void testGetInitialBalance() {
        assertEquals(1000, account1.getBalance(), 0.01);
        assertEquals(500, account2.getBalance(), 0.01);
    }

    @Test
    void testDeposit() {
        account1.deposit(500);
        assertEquals(1500, account1.getBalance(), 0.01);
    }

    @Test
    void testWithdraw() {
        account1.withdraw(600);
        assertEquals(400, account1.getBalance(), 0.01);
    }

    @Test
    void testTransferTo_Success() {
        account1.transferTo(account2, 300);
        assertEquals(800, account1.getBalance(), 0.01);
        assertEquals(800, account2.getBalance(), 0.01);
    }

    @Test
    void testTransferTo_InsufficientFunds() {
        Exception exception = assertThrows(IllegalArgumentException.class, () -> account1.transferTo(account2, 2000));
        assertTrue(exception.getMessage().contains("insufficient funds"));
    }
}
```

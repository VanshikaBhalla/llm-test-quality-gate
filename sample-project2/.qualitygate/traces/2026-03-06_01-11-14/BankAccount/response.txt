 Here is the JUnit 5 test class for the BankAccount class:

```java
package com.example;

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.*;

class BankAccountTest {

    private BankAccount account1;
    private BankAccount account2;
    private BankAccount account3;

    @BeforeEach
    void setUp() {
        account1 = new BankAccount("123456789", 1000);
        account2 = new BankAccount("234567890", 500);
        account3 = new BankAccount("345678901", 0);
    }

    @Test
    void testGetAccountNumber() {
        assertEquals("123456789", account1.getAccountNumber());
        assertEquals("234567890", account2.getAccountNumber());
        assertEquals("345678901", account3.getAccountNumber());
    }

    @Test
    void testGetBalance() {
        assertEquals(1000, account1.getBalance(), 0.01);
        assertEquals(500, account2.getBalance(), 0.01);
        assertEquals(0, account3.getBalance(), 0.01);
    }

    @Test
    void testDeposit() {
        account1.deposit(500);
        assertEquals(1500, account1.getBalance(), 0.01);
    }

    @Test
    void testWithdraw() {
        account1.withdraw(300);
        assertEquals(700, account1.getBalance(), 0.01);
    }

    @Test
    void testTransferTo() {
        account1.transferTo(account2, 200);
        assertEquals(800, account1.getBalance(), 0.01);
        assertEquals(700, account2.getBalance(), 0.01);
    }

    @Test
    void testTransferInsufficientFunds() {
        account1.transferTo(account2, 2000);
        assertEquals(1000, account1.getBalance(), 0.01);
        assertEquals(500, account2.getBalance(), 0.01);
    }

}
```

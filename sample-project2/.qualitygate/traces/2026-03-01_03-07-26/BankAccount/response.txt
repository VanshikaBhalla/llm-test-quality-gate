 Here is the JUnit 5 test class for the `BankAccount` class with the specified rules:

```java
package com.example;

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.*;

class BankAccountTest {

    private BankAccount account1;
    private BankAccount account2;

    @BeforeEach
    void setUp() {
        account1 = new BankAccount("123456789");
        account2 = new BankAccount("098765432");
        account1.deposit(100);
        account2.deposit(200);
    }

    @Test
    void testGetAccountNumber() {
        assertEquals("123456789", account1.getAccountNumber());
        assertEquals("098765432", account2.getAccountNumber());
    }

    @Test
    void testGetBalance() {
        assertEquals(100, account1.getBalance(), 0.01);
        assertEquals(200, account2.getBalance(), 0.01);
    }

    @Test
    void testDeposit() {
        account1.deposit(50);
        assertEquals(150, account1.getBalance(), 0.01);
    }

    @Test
    void testWithdraw() {
        account1.withdraw(20);
        assertEquals(80, account1.getBalance(), 0.01);
        assertThrows(IllegalArgumentException.class, () -> account1.withdraw(300));
    }

    @Test
    void testTransferTo() {
        account1.transferTo(account2, 50);
        assertEquals(150, account1.getBalance(), 0.01);
        assertEquals(250, account2.getBalance(), 0.01);
        assertThrows(IllegalArgumentException.class, () -> account1.transferTo(account1, 300));
    }
}
```

This test class creates a `BankAccount` for two accounts and initializes them with deposits. It then tests the `getAccountNumber()`, `getBalance()`, `deposit()`, `withdraw()`, and `transferTo()` methods with valid and invalid inputs.

 Here is a simple JUnit 5 test class for the `BankAccount` class:

```java
package com.example;

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.*;

class BankAccountTest {

    private BankAccount account1;
    private BankAccount account2;

    @BeforeEach
    void setUp() {
        account1 = new BankAccount("123456789", 100.0);
        account2 = new BankAccount("987654321", 50.0);
    }

    @Test
    void testGetAccountNumber() {
        assertEquals("123456789", account1.getAccountNumber());
        assertEquals("987654321", account2.getAccountNumber());
    }

    @Test
    void testGetInitialBalance() {
        assertEquals(100.0, account1.getBalance());
        assertEquals(50.0, account2.getBalance());
    }

    @Test
    void testDeposit() {
        account1.deposit(50);
        assertEquals(150.0, account1.getBalance());
    }

    @Test
    void testWithdraw() {
        account1.withdraw(20);
        assertEquals(80.0, account1.getBalance());
    }

    @Test
    void testTransferTo_SufficientFunds() {
        account1.transferTo(account2, 30);
        assertEquals(70.0, account1.getBalance());
        assertEquals(80.0, account2.getBalance());
    }

    @Test
    void testTransferTo_InsufficientFunds() {
        assertThrows(IllegalArgumentException.class, () -> account1.transferTo(account2, 200));
    }
}
```

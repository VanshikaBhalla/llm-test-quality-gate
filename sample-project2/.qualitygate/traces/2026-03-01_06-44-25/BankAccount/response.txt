 Here is the JUnit 5 test class for the `BankAccount`:

```java
package com.example;

import org.junit.jupiter.api.Assertions;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

public class BankAccountTest {

    private BankAccount account1;
    private BankAccount account2;

    @BeforeEach
    public void setup() {
        account1 = new BankAccount("123456789", 100);
        account2 = new BankAccount("987654321", 50);
    }

    @Test
    public void testGetAccountNumber() {
        Assertions.assertEquals("123456789", account1.getAccountNumber());
        Assertions.assertEquals("987654321", account2.getAccountNumber());
    }

    @Test
    public void testGetBalance() {
        Assertions.assertEquals(100, account1.getBalance(), 0.01);
        Assertions.assertEquals(50, account2.getBalance(), 0.01);
    }

    @Test
    public void testDeposit() {
        account1.deposit(50);
        Assertions.assertEquals(150, account1.getBalance(), 0.01);
    }

    @Test
    public void testWithdraw() {
        account1.withdraw(20);
        Assertions.assertEquals(80, account1.getBalance(), 0.01);
    }

    @Test
    public void testTransferTo_Success() {
        account1.transferTo(account2, 30);
        AccountNumberAndBalance account1Data = new AccountNumberAndBalance(account1.getAccountNumber(), account1.getBalance());
        AccountNumberAndBalance account2Data = new AccountNumberAndBalances(account2.getAccountNumber(), account2.getBalance());
        Assertions.assertEquals(new AccountNumberAndBalance("123456789", 70), account1Data);
        Assertions.assertEquals(new AccountNumberAndBalances("987654321", 80), account2Data);
    }

    @Test
    public void testTransferTo_InsufficientFunds() {
        Exception exception = Assertions.assertThrows(IllegalArgumentException.class, () -> account1.transferTo(account2, 151));
        Assertions.assertEquals("Insufficient funds", exception.getMessage());
    }

    private record AccountNumberAndBalance(String accountNumber, double balance) { }
    private record AccountNumberAndBalances(String accountNumber, double balance) { }
}
```

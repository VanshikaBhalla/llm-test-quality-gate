 Here's the JUnit 5 test class for the BankAccount:

```java
package com.example;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertThrows;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

class BankAccountTest {
    private BankAccount account1;
    private BankAccount account2;

    @BeforeEach
    void setUp() {
        account1 = new BankAccount("123456789", 100);
        account2 = new BankAccount("987654321", 50);
    }

    @Test
    void testGetAccountNumber() {
        assertEquals("123456789", account1.getAccountNumber());
        assertEquals("987654321", account2.getAccountNumber());
    }

    @Test
    void testGetBalance() {
        assertEquals(100, account1.getBalance(), 0.0);
        assertEquals(50, account2.getBalance(), 0.0);
    }

    @Test
    void testDeposit() {
        account1.deposit(50);
        assertEquals(150, account1.getBalance(), 0.0);
    }

    @Test
    void testWithdraw() {
        account1.withdraw(25);
        assertEquals(75, account1.getBalance(), 0.0);
        assertThrows(IllegalArgumentException.class, () -> account1.withdraw(-5));
    }

    @Test
    void testTransferTo() {
        account1.transferTo(account2, 30);
        assertEquals(70, account1.getBalance(), 0.0);
        assertEquals(80, account2.getBalance(), 0.0);
        assertThrows(IllegalArgumentException.class, () -> account1.transferTo(account2, -5));
    }
}
```

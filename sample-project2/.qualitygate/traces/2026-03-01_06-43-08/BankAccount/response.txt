 Here's a JUnit 5 test class for the `BankAccount` class:

```java
package com.example;

import static org.junit.jupiter.api.Assertions.*;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

class BankAccountTest {

    private BankAccount account1;
    private BankAccount account2;

    @BeforeEach
    void setUp() {
        account1 = new BankAccount("123456789", 100);
        account2 = new BankAccount("987654321", 200);
    }

    @Test
    void testGetAccountNumber() {
        assertEquals("123456789", account1.getAccountNumber());
        assertEquals("987654321", account2.getAccountNumber());
    }

    @Test
    void testInitialBalance() {
        assertEquals(100, account1.getBalance());
        assertEquals(200, account2.getBalance());
    }

    @Test
    void testDeposit() {
        account1.deposit(50);
        assertEquals(150, account1.getBalance());
    }

    @Test
    void testWithdraw() {
        account1.withdraw(75);
        assertEquals(25, account1.getBalance(), "Account 1 balance should be 25");
        account2.withdraw(101);
        assertEquals(0, account2.getBalance(), "Account 2 balance should be 0");
    }

    @Test
    void testTransferTo() {
        account1.transferTo(account2, 50);
        assertEquals(75, account1.getBalance(), "Account 1 balance after transfer should be 75");
        assertEquals(250, account2.getBalance(), "Account 2 balance after transfer should be 250");
    }
}
```

 Here is the JUnit 5 test class for the `BankAccount` class:

```java
package com.example;

import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.Assertions;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

public class BankAccountTest {

    private BankAccount account1;
    private BankAccount account2;

    @BeforeEach
    public void setup() {
        account1 = new BankAccount("123456789", 1000);
        account2 = new BankAccount("987654321", 500);
    }

    @AfterEach
    public void teardown() {
        account1 = null;
        account2 = null;
    }

    @Test
    public void testGetAccountNumber() {
        Assertions.assertEquals("123456789", account1.getAccountNumber());
        Assertions.assertEquals("987654321", account2.getAccountNumber());
    }

    @Test
    public void testGetBalance() {
        Assertions.assertEquals(1000, account1.getBalance(), 0.01);
        Assertions.assertEquals(500, account2.getBalance(), 0.01);
    }

    @Test
    public void testDeposit() {
        account1.deposit(500);
        Assertions.assertEquals(1500, account1.getBalance(), 0.01);

        account2.deposit(200);
        Assertions.assertEquals(700, account2.getBalance(), 0.01);
    }

    @Test
    public void testWithdraw() {
        account1.withdraw(200);
        Assertions.assertEquals(800, account1.getBalance(), 0.01);

        account2.withdraw(300);
        Assertions.assertEquals(200, account2.getBalance(), 0.01);
    }

    @Test
    public void testTransferTo() {
        account1.transferTo(account2, 300);
        Assertions.assertEquals(700, account1.getBalance(), 0.01);
        Assertions.assertEquals(800, account2.getBalance(), 0.01);
    }
}
```

This test class creates two `BankAccount` instances, tests the getters, and tests depositing, withdrawing, and transferring money between the accounts. It ensures that the balance and account numbers are set correctly after each operation.

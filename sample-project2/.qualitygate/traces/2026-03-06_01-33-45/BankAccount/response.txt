 Here is the JUnit 5 test class for the `BankAccount` class:

```java
package com.example;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertThrows;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

class BankAccountTest {

    private BankAccount account1;
    private BankAccount account2;

    @BeforeEach
    void setup() {
        account1 = new BankAccount("123456789", 100.0);
        account2 = new BankAccount("098765432", 50.0);
    }

    @Test
    void testGetAccountNumber() {
        assertEquals("123456789", account1.getAccountNumber());
        assertEquals("098765432", account2.getAccountNumber());
    }

    @Test
    void testGetBalance() {
        assertEquals(100.0, account1.getBalance(), 0.01);
        assertEquals(50.0, account2.getBalance(), 0.01);
    }

    @Test
    void testDeposit() {
        account1.deposit(50.0);
        assertEquals(150.0, account1.getBalance(), 0.01);
    }

    @Test
    void testWithdraw() {
        account1.withdraw(50.0);
        assertEquals(50.0, account1.getBalance(), 0.01);
    }

    @Test
    void testTransferTo() {
        account1.transferTo(account2, 25.0);
        assertEquals(75.0, account1.getBalance(), 0.01);
        assertEquals(75.0, account2.getBalance(), 0.01);
    }

    @Test
    void testTransferInsufficientFunds() {
        assertThrows(IllegalArgumentException.class, () -> account1.transferTo(account2, 200.0));
    }

    @Test
    void testTransferNullAccount() {
        assertThrows(NullPointerException.class, () -> account1.transferTo(null, 25.0));
    }
}
```

 Here is the JUnit 5 test class for BankAccount:

```java
package com.example;

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.*;

class BankAccountTest {
    private BankAccount account1;
    private BankAccount account2;

    @BeforeEach
    void setup() {
        account1 = new BankAccount("123456789", 100);
        account2 = new BankAccount("987654321", 200);
    }

    @Test
    void testGetAccountNumber() {
        assertEquals("123456789", account1.getAccountNumber());
        assertEquals("987654321", account2.getAccountNumber());
    }

    @Test
    void testGetBalance() {
        assertEquals(100, account1.getBalance(), 0.01);
        assertEquals(200, account2.getBalance(), 0.01);
    }

    @Test
    void testDeposit() {
        account1.deposit(50);
        assertEquals(150, account1.getBalance(), 0.01);
    }

    @Test
    void testWithdraw() {
        account2.withdraw(100);
        assertEquals(100, account2.getBalance(), 0.01);
    }

    @Test
    void testTransferTo_SufficientFunds() {
        account1.transferTo(account2, 50);
        assertEquals(150, account1.getBalance(), 0.01);
        assertEquals(250, account2.getBalance(), 0.01);
    }

    @Test
    void testTransferTo_InsufficientFunds() {
        account1.transferTo(account2, 300);
        assertEquals(100, account1.getBalance(), 0.01);
        assertEquals(200, account2.getBalance(), 0.01);
    }
}
```

 Here is the JUnit 5 test class for the `BankAccount` class:

```java
package com.example;

import static org.junit.jupiter.api.Assertions.*;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

class BankAccountTest {
    private BankAccount account1;
    private BankAccount account2;

    @BeforeEach
    void setUp() {
        account1 = new BankAccount("123456789", 100);
        account2 = new BankAccount("987654321", 50);
    }

    @Test
    void testGetters() {
        assertEquals("123456789", account1.getAccountNumber());
        assertEquals(100, account1.getBalance(), 0.001);
        assertEquals("987654321", account2.getAccountNumber());
        assertEquals(50, account2.getBalance(), 0.001);
    }

    @Test
    void testDeposit() {
        account1.deposit(50);
        assertEquals(150, account1.getBalance(), 0.001);
    }

    @Test
    void testWithdraw() {
        account1.withdraw(30);
        assertEquals(70, account1.getBalance(), 0.001);
    }

    @Test
    void testTransferTo_sufficientFunds() {
        account1.transferTo(account2, 25);
        assertEquals(75, account1.getBalance(), 0.001);
        assertEquals(75 + 50, account2.getBalance(), 0.001);
    }

    @Test
    void testTransferTo_insufficientFunds() {
        account1.transferTo(account2, 1000);
        assertEquals(100, account1.getBalance(), 0.001);
        assertEquals(50, account2.getBalance(), 0.001);
    }
}
```

 Here is the JUnit 5 test class for the Calculator:

```java
package com.example;

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.*;

class CalculatorTest {
    private Calculator calculator;

    @BeforeEach
    void setup() {
        this.calculator = new Calculator();
    }

    @Test
    void testAdd() {
        assertEquals(5, this.calculator.add(2, 3));
        assertEquals(-1, this.calculator.add(-2, 3));
        assertEquals(0, this.calculator.add(-2, -3));
    }

    @Test
    void testDivide() {
        assertEquals(2, this.calculator.divide(4, 2));
        assertThrows(ArithmeticException.class, () -> this.calculator.divide(4, 0));
    }
}
```

 Here's the JUnit 5 test class for the Calculator:

```java
package com.example;

import org.junit.jupiter.api.Assertions;
import org.junit.jupiter.api.Test;

public class CalculatorTest {
    private Calculator calculator = new Calculator();

    @Test
    public void testAdd() {
        Assertions.assertEquals(5, calculator.add(2, 3));
        Assertions.assertEquals(10, calculator.add(4, 6));
        Assertions.assertEquals(0, calculator.add(-2, 2));
    }

    @Test
    public void testDivide() {
        Assertions.assertEquals(2, calculator.divide(4, 2));
        Assertions.assertThrows(ArithmeticException.class, () -> calculator.divide(4, 0));
        Assertions.assertEquals(3, calculator.divide(12, 4));
    }
}
```

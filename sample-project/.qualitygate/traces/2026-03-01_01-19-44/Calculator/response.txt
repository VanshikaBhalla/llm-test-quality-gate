 Here's a JUnit 5 test class for the `Calculator` with the specified rules:

```java
package com.example;

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.Test;

public class CalculatorTest {

    private final Calculator calculator = new Calculator();

    @Test
    public void testAdd() {
        assertEquals(5, calculator.add(2, 3);
        assertEquals(10, calculator.add(4, 6);
        assertEquals(-1, calculator.add(-2, 3));
    }

    @Test
    public void testDivide() {
        assertEquals(2, calculator.divide(4, 2);
        assertEquals(0, calculator.divide(4, 0); // Ensure divide by zero exception is thrown
        assertThrows(ArithmeticException.class, () -> calculator.divide(4, 0));
    }
}
```
